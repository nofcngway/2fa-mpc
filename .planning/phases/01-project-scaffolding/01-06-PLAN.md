**mpc/docker-compose.yaml (per D-03, D-05 -- single PostgreSQL, 3 node instances via config):**
```yaml
services:
  mpc-postgres:
    image: postgres:17
    environment:
      POSTGRES_DB: mpc_db
      POSTGRES_USER: mpc_user
      POSTGRES_PASSWORD: mpc_pass
    ports:
      - "5435:5432"
    volumes:
      - mpc-pg-data:/var/lib/postgresql/data

  mpc-kafka:
    image: bitnami/kafka:4.1
    environment:
      KAFKA_CFG_NODE_ID: 1
      KAFKA_CFG_PROCESS_ROLES: broker,controller
      KAFKA_CFG_CONTROLLER_QUORUM_VOTERS: 1@mpc-kafka:9093
      KAFKA_CFG_LISTENERS: PLAINTEXT://:9092,CONTROLLER://:9093
      KAFKA_CFG_ADVERTISED_LISTENERS: PLAINTEXT://localhost:9094
      KAFKA_CFG_CONTROLLER_LISTENER_NAMES: CONTROLLER
      KAFKA_CFG_LISTENER_SECURITY_PROTOCOL_MAP: CONTROLLER:PLAINTEXT,PLAINTEXT:PLAINTEXT

      - "9094:9092"

  mpc-pg-data:
```
