**Config loader (per INFRA-08):**

**auth/config/config.go:**
```go
package config

import (
    "os"
    "time"
    "gopkg.in/yaml.v3"
)

type Config struct {
    Server   ServerConfig   `yaml:"server"`
    Database DatabaseConfig `yaml:"database"`
    Redis    RedisConfig    `yaml:"redis"`
    Kafka    KafkaConfig    `yaml:"kafka"`
    JWT      JWTConfig      `yaml:"jwt"`
}

type ServerConfig struct {
    Port int `yaml:"port"`
}

type DatabaseConfig struct {
    DSN string `yaml:"dsn"`
}

type RedisConfig struct {
    Addr     string `yaml:"addr"`
    Password string `yaml:"password"`
    DB       int    `yaml:"db"`
}

type KafkaConfig struct {
    Brokers []string `yaml:"brokers"`
    Topic   string   `yaml:"topic"`
}

type JWTConfig struct {
    PrivateKeyPath  string        `yaml:"private_key_path"`
    PublicKeyPath   string        `yaml:"public_key_path"`
    AccessTokenTTL  time.Duration `yaml:"access_token_ttl"`
    RefreshTokenTTL time.Duration `yaml:"refresh_token_ttl"`
}

func Load(path string) (*Config, error) {
    data, err := os.ReadFile(path)
    if err != nil {
        return nil, err
    }
    var cfg Config
    if err := yaml.Unmarshal(data, &cfg); err != nil {
        return nil, err
    }
    return &cfg, nil
}
```

```go
package bootstrap

import (

    "google.golang.org/grpc/health"
    healthpb "google.golang.org/grpc/health/grpc_health_v1"
)

// NewPGStorage creates PostgreSQL connection pool.
func NewPGStorage(ctx context.Context, cfg *config.Config) (*pgstorage.PGStorage, error) {
    storage, err := pgstorage.New(ctx, cfg.Database.DSN)
    if err != nil {
        return nil, err
    }
    slog.Info("PostgreSQL connected")
    return storage, nil
}

// NewRedisStorage creates Redis client.
func NewRedisStorage(cfg *config.Config) (*redisstorage.RedisStorage, error) {
    rs := redisstorage.New(cfg.Redis.Addr, cfg.Redis.Password, cfg.Redis.DB)
    slog.Info("Redis connected")
    return rs, nil
}

// NewAuthService creates service layer.
func NewAuthService(storage *pgstorage.PGStorage, redis *redisstorage.RedisStorage) *authService.AuthService {
    return authService.NewAuthService(storage, redis)
}

// NewAuthServiceAPI creates gRPC handler layer.
func NewAuthServiceAPI(service *authService.AuthService) *auth_service_api.AuthServiceAPI {
    return auth_service_api.NewAuthServiceAPI(service)
}

// NewGRPCServer creates and configures gRPC server with interceptors and service registration.
func NewGRPCServer(api *auth_service_api.AuthServiceAPI) *grpc.Server {
    server := grpc.NewServer(
        grpc.UnaryInterceptor(middleware.LoggingInterceptor),
    )
    pb.RegisterAuthServiceServer(server, api)

    // Health check
    healthServer := health.NewServer()
    healthpb.RegisterHealthServer(server, healthServer)
    healthServer.SetServingStatus("", healthpb.HealthCheckResponse_SERVING)

    return server
}
```

```go
package main

import (
    "context"
    "fmt"
    "log/slog"
    "net"
    "os"
    "os/signal"
    "syscall"

    "github.com/vbncursed/vkr/auth/config"
    "github.com/vbncursed/vkr/auth/internal/bootstrap"
)

func main() {
    // Load config
    cfg, err := config.Load("config.yaml")
    if err != nil {
        slog.Error("failed to load config", "error", err)
        os.Exit(1)
    }

    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    // Bootstrap dependencies
    storage, err := bootstrap.NewPGStorage(ctx, cfg)
    if err != nil {
        slog.Error("failed to connect to PostgreSQL", "error", err)
        os.Exit(1)
    }
    defer storage.Close()

    redisStorage, err := bootstrap.NewRedisStorage(cfg)
    if err != nil {
        slog.Warn("failed to connect to Redis, continuing without", "error", err)
        // Per D-10: log warning, don't exit
    }

    service := bootstrap.NewAuthService(storage, redisStorage)
    api := bootstrap.NewAuthServiceAPI(service)
    grpcServer := bootstrap.NewGRPCServer(api)

    // Start listener
    lis, err := net.Listen("tcp", fmt.Sprintf(":%d", cfg.Server.Port))
    if err != nil {
        slog.Error("failed to listen", "error", err, "port", cfg.Server.Port)
        os.Exit(1)
    }

    // Graceful shutdown
    sigCh := make(chan os.Signal, 1)
    signal.Notify(sigCh, syscall.SIGINT, syscall.SIGTERM)

    go func() {
        <-sigCh
        slog.Info("shutting down...")
        grpcServer.GracefulStop()
        cancel()
    }()

    slog.Info("auth service started", "port", cfg.Server.Port)
    if err := grpcServer.Serve(lis); err != nil {
        slog.Error("gRPC server error", "error", err)
        os.Exit(1)
    }
}
```
